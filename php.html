üîπ Conditional Statements

<?php
  $x = 10;

  if ($x > 5) {
    echo "x is greater than 5";
  }
?>
<?php
  $x = 10;

  if ($x > 15) {
    echo "x is greater than 15";
  } else {
    echo "x is not greater than 15";
  }
?>
<?php
  $x = 10;

  if ($x > 15) {
    echo "x is greater than 15";
  } elseif ($x > 5) {
    echo "x is greater than 5";
  } else {
    echo "x is 5 or less";
  }
?>
<?php
  $color = "red";

  switch ($color) {
    case "red":
      echo "Color is red";
      break;
    case "blue":
      echo "Color is blue";
      break;
    default:
      echo "Color is unknown";
  }
?>

üîπ Loops in PHP

<?php
  $x = 1;
  while($x <= 5) {
    echo "The number is: $x <br>";
    $x++;
  }
?>
<?php
  for ($x = 0; $x <= 5; $x++) {
    echo "The number is: $x <br>";
  }
?>
<?php
  $colors = array("red", "green", "blue");
  foreach ($colors as $value) {
    echo "$value <br>";
  }
?>
<?php
  $x = 1;
  do {
    echo "The number is: $x <br>";
    $x++;
  } while ($x <= 5);
?>
